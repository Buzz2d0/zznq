cel 是建立 protocol 之上的。

在 demo 项目下创建 fprint 包；并在其目录下创建 fprint.proto 文件

syntax = "proto3";
package fprint; // https://developers.google.com/protocol-buffers/docs/proto3#packages

option go_package = "fprint/bp"; // golang package 包名

message Response {
  string url = 1;
  int32 status = 2 ;
  map<string, string> headers = 3;
  string content_type = 4;
  string title = 5;
  bytes body = 6;
  string md5 = 7;
}

使用 $ protoc --go_out=./ fprint/fprint.proto 生成 go 代码

cel.go:

package fprint

import (
	"bytes"
	"log"
	"regexp"

	"github.com/google/cel-go/cel"
	"github.com/google/cel-go/checker/decls"
	"github.com/google/cel-go/common/types"
	"github.com/google/cel-go/common/types/ref"
	"github.com/google/cel-go/interpreter/functions"

	"demo/fprint/bp"

	exprpb "google.golang.org/genproto/googleapis/api/expr/v1alpha1"
)

func NewEnv(f *fprintLib) (*cel.Env, error) {
	return cel.NewEnv(cel.Lib(f))
}

func Evaluate(env *cel.Env, expression string, params map[string]interface{}) (ref.Val, error) {
	ast, iss := env.Compile(expression)
	if iss.Err() != nil {
		log.Println("compile error")
		return nil, iss.Err()
	}

	prg, err := env.Program(ast)
	if err != nil {
		log.Println("Program creation error")
		return nil, err
	}

	out, _, err := prg.Eval(params)
	if err != nil {
		log.Println("Evaluation error")
		return nil, err
	}
	return out, nil
}

type fprintLib struct {
	envOptions     []cel.EnvOption
	programOptions []cel.ProgramOption
}

func NewEnvOption() *fprintLib {
	f := &fprintLib{}
	f.envOptions = []cel.EnvOption{
        // 注册 response 类型
		cel.Container("fprint.Response"),
		cel.Types(
			&bp.Response{},
		),
		cel.Declarations(
			decls.NewIdent("response", decls.NewObjectType("fprint.Response"), nil),
		),
        // 定义方法、函数等
		cel.Declarations(
			decls.NewFunction("bcontains",
				decls.NewInstanceOverload("bytes_bcontains_bytes",
					[]*exprpb.Type{decls.Bytes, decls.Bytes},
					decls.Bool)),
			decls.NewFunction("regexp",
				decls.NewOverload("regexp_string_string",
					[]*exprpb.Type{decls.String, decls.String},
					decls.Bool)),
			decls.NewFunction("version",
				decls.NewOverload("version_string_string_int",
					[]*exprpb.Type{decls.String, decls.String, decls.Int},
					decls.String)),
		),
	}
	f.programOptions = []cel.ProgramOption{
        cel.Functions(
            // bytes_bcontains_bytes 的实现
			&functions.Overload{
				Operator: "bytes_bcontains_bytes",
				Binary: func(lhs ref.Val, rhs ref.Val) ref.Val {
					v1, ok := lhs.(types.Bytes)
					if !ok {
						return types.ValOrErr(lhs, "unexpected type '%v' passed to bcontains", lhs.Type())
					}
					v2, ok := rhs.(types.Bytes)
					if !ok {
						return types.ValOrErr(rhs, "unexpected type '%v' passed to bcontains", rhs.Type())
					}
					return types.Bool(bytes.Contains(v1, v2))
				},
			},
            // regexp_string_string 的实现
			&functions.Overload{
				Operator: "regexp_string_string",
				Binary: func(arg1, arg2 ref.Val) ref.Val {
					reg, ok := arg1.(types.String)
					if !ok {
						return types.ValOrErr(arg1, "unexpected type '%v' passed to regexp", arg1.Type())
					}
					val, ok := arg2.(types.String)
					if !ok {
						return types.ValOrErr(arg2, "unexpected type '%v' passed to regexp", arg2.Type())
					}
					matched, err := regexp.MatchString(string(reg), string(val))
					if err != nil {
						return types.ValOrErr(arg2, "Wrong regular syntax '%v' passed to version", arg2.Type())
					}
					return types.Bool(matched)
				},
			},
            // version_string_string_int 的实现
			&functions.Overload{
				Operator: "version_string_string_int",
				Function: func(values ...ref.Val) ref.Val {
					if len(values) != 3 {
						return types.NewErr("too many arguments to 'substr'")
					}
					offset, ok := values[2].(types.Int)
					if !ok {
						return types.ValOrErr(values[2], "unexpected type '%v' passed to version", values[2].Type())
					}
					if offset < 0 {
						offset = 0
					}

					reg, ok := values[0].(types.String)
					if !ok {
						return types.ValOrErr(values[0], "unexpected type '%v' passed to version", values[0].Type())
					}
					val, ok := values[1].(types.String)
					if !ok {
						return types.ValOrErr(values[1], "unexpected type '%v' passed to version", values[1].Type())
					}
					regx, err := regexp.Compile(string(reg))
					if err != nil {
						return types.ValOrErr(values[0], "Wrong regular syntax '%v' passed to version", values[0].Type())
					}
					vals := regx.FindStringSubmatch(string(val))
					if len(vals) <= 0 {
						return types.String("")
					}
					offset += 1
					if int(offset) > len(vals) {
						offset = types.Int(len(vals) - 1)
					}
					return types.String(vals[offset])
				},
			},
		),
	}
	return f
}

// EnvOptions returns options for the standard CEL function declarations and macros.
func (f *fprintLib) CompileOptions() []cel.EnvOption {
	return f.envOptions
}

// ProgramOptions returns function implementations for the standard CEL functions.
func (f *fprintLib) ProgramOptions() []cel.ProgramOption {
	return f.programOptions
}

写个单元测试运行下，cel_test.go ：

package fprint

import (
	"demo/fprint/bp"
	"testing"
)

func TestFprint(t *testing.T) {
	flib := NewEnvOption()
	env, err := NewEnv(flib)
	if err != nil {
		t.Fatal(err)
	}
	params := map[string]interface{}{}
	params["response"] = &bp.Response{
		Url:    "http://www.baidu.com",
		Status: 404,
		Headers: map[string]string{
			"server": "nginx",
		},
		ContentType: "application/json",
		Title:       "this is a titleTest Bootstrap v0.1.1 test",
		Body:        []byte("http.body"),
	}

	out, err := Evaluate(env, `response.status == 200`, params)
	if err != nil {
		t.Fatal(err)
	}
	t.Log("response.status == 200", out)

	out, err = Evaluate(env, `response.headers["server"] == "nginx"`, params)
	if err != nil {
		t.Fatal(err)
	}
	t.Log(`response.headers["server"] == "nginx"`, out)

	out, err = Evaluate(env, `response.body.bcontains(b"body")`, params)
	if err != nil {
		t.Fatal(err)
	}
	t.Log(`response.body.bcontains(b"body")`, out)

	out, err = Evaluate(env, `regexp("(?im)bootstrap ([vV])?([0-9\\.]+)", response.title)`, params)
	if err != nil {
		t.Fatal(err)
	}
	t.Log(`regexp("(?im)bootstrap ([vV])?([0-9\\.]+)", response.title)`, out)

	out, err = Evaluate(env, `version("(?im)bootstrap ([vV])?([0-9\\.]+)", response.title, 1)`, params)
	if err != nil {
		t.Fatal(err)
	}
	t.Log(`version("(?im)bootstrap ([vV])?([0-9\\.]+)", response.title, 1)`, out)
}
